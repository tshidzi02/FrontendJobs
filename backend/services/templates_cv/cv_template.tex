% ── Section heading ───────────────────────────────────────────────────────────
% Uses only \colorbox and \parbox — zero TikZ, zero fragile commands.
% Works on every MiKTeX/TeX Live installation without extra libraries.
\newcommand{\cvsection}[1]{%
  \vspace{6pt}%
  \noindent%
  \colorbox{DKGreen}{\hspace{4pt}\rule{0pt}{14pt}\hspace{4pt}}%
  \colorbox{CardBG}{%
    \parbox[c][14pt]{\dimexpr\linewidth-16pt\relax}{%
      \hspace{8pt}\textcolor{DKGreen}{\textbf{\large\MakeUppercase{#1}}}%
    }%
  }%
  \par
  \noindent{\color{BorderClr}\rule{\linewidth}{0.6pt}}%
  \vspace{4pt}\par\noindent%
}

% ── Skill card ────────────────────────────────────────────────────────────────
\newcommand{\skillcard}[2]{%
  \begin{mdframed}[
    linecolor=BorderClr, linewidth=0.6pt,
    leftline=true, rightline=true, topline=true, bottomline=true,
    innerleftmargin=7pt, innerrightmargin=6pt,
    innertopmargin=5pt,  innerbottommargin=5pt,
    backgroundcolor=white,
    leftmargin=0pt, rightmargin=0pt,
    skipabove=4pt,  skipbelow=4pt,
  ]%
  \textcolor{DKGreen}{\textbf{#1}}\\[2pt]%
  {\small\textcolor{LiteTxt}{#2}}%
  \end{mdframed}%
}

% ── Language bar ──────────────────────────────────────────────────────────────
% Pure \ifnum — no packages needed, works everywhere
\newcommand{\sqon}{\textcolor{DKGreen}{\rule{8pt}{8pt}}\,}
\newcommand{\sqoff}{\textcolor{BorderClr}{\rule{8pt}{8pt}}\,}
\newcommand{\langbar}[1]{%
  \ifnum#1>0\sqon\else\sqoff\fi%
  \ifnum#1>1\sqon\else\sqoff\fi%
  \ifnum#1>2\sqon\else\sqoff\fi%
  \ifnum#1>3\sqon\else\sqoff\fi%
  \ifnum#1>4\sqon\else\sqoff\fi%
  \ifnum#1>5\sqon\else\sqoff\fi%
}

% ── Thin rule ─────────────────────────────────────────────────────────────────
